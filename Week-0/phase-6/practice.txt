# TOTAL SPEND
customer_spend = df.groupBy("customer_id", "city") \
    .agg(F.sum("amount").alias("total_spend"))

# TOP 3 CUSTOMERS PER CITY
window_city = Window.partitionBy("city").orderBy(F.col("total_spend").desc())
top3 = customer_spend.withColumn("rank", F.row_number().over(window_city)) \
    .filter(F.col("rank") <= 3)
top3.show()

# RUNNING TOTAL
window_run = Window.orderBy("date")
running = df.withColumn(
    "running_total",
    F.sum("amount").over(window_run)
)
running.show()

# CUSTOMER RANKING
ranked = customer_spend.withColumn(
    "rank",
    F.dense_rank().over(Window.orderBy(F.col("total_spend").desc()))
)
ranked.show()

# LAG FUNCTION
lag_df = df.withColumn(
    "prev_amount",
    F.lag("amount").over(window_run)
)
lag_df.show()
